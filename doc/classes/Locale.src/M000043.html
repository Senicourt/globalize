  def self.country(code, variant=:formal, locale=nil)
    code = ( (MLR_ISO3166_CODE || DEFAULT_MLR_ISO3166_CODE).to_s == 'numeric' ? code.to_i : code.to_s )
    country = countries(locale || current)[code]
    country.nil? ? nil : country[variant]
  end
