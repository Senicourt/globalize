  def strftime(str)
    str.gsub(/%./) do |m|
      case m
        when '%a'
          Locale.abday(self.wday)
        when '%A'
          Locale.day(self.wday)
        when '%b'
          Locale.abmonth(self.month)
        when '%B'
          Locale.month(self.month)
        else
          strftime_orig(m)
      end
    end
  end
